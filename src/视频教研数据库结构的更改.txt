    confId  varchar(32) Null             视频会议的ID
    confKey varchar(32) Null             视频会议的Key
    hostUrl varchar(512)    Null         视频会议的管理Url
    attendeeUrl varchar(512)    Null     视频会议的参会Url
    subject nvarchar(512)   Null         会议主题
    startTime   Date Null                会议开始时间
    endTime Date Null                    会议结束时间
    hostName    varchar(50) Null         主持人(应该是存在的用户)
    openType    bit Null                 true 公开   false 不公开
    passwd  varchar(32) Null             会议室密码
    agenda  nvarchar(1024)  Null         会议描述
    beforehandTime  int Null             提前进入会议室时间(分钟?)
    webBaseUrl  varchar(512)    Null     网络视频会议的网址
    conferenceType  int Null             会议类型  预约会议：0 即时会议：1 固定会议：2 周期会议：3
    href varchar(512)    Null            网络视频会议的视频文件
    flvThumbNailHref varchar(512)    Null    网络视频会议的视频文件，缩略图
    flvHref varchar(512)    Null            网络视频会议的视频文件转换为flv文件
    attendeeAmount int Null              参会人数
    attendeeIds                             参会人Id
    attendeeNames                            参会人
    status              int  状态
    resourceIds varchar(1024)   Null     ----资源文件
